JavaScript Component Library

// REUSABLE COMPONENT LIBRARY
const AnchorStack = {
    // Create a stat box
    createStatBox: (number, label, detail) => {
        return `
            <div class="stat-box">
                <div class="stat-number">${number}</div>
                <p class="stat-label">${label}</p>
                <p class="stat-detail">${detail}</p>
            </div>
        `;
    },

    // Create an interactive card
    createCard: (title, content, clickable = false) => {
        const cardClass = clickable ? 'card-interactive' : 'card-static';
        return `
            <div class="${cardClass}">
                <h3 class="font-bold mb-2" style="color: var(--color-primary)">${title}</h3>
                <p>${content}</p>
            </div>
        `;
    },

    // Initialize tab navigation
    initTabs: () => {
        const tabs = document.querySelectorAll('.nav-tab');
        const sections = document.querySelectorAll('.content-section');
        
        tabs.forEach(tab => {
            tab.addEventListener('click', () => {
                const target = tab.dataset.target;
                
                tabs.forEach(t => t.classList.remove('active'));
                sections.forEach(s => s.classList.remove('active'));
                
                tab.classList.add('active');
                document.getElementById(target)?.classList.add('active');
            });
        });
    },

    // Smooth scroll to element
    scrollToElement: (elementId) => {
        document.getElementById(elementId)?.scrollIntoView({ 
            behavior: 'smooth', 
            block: 'start' 
        });
    }
};

  • ✓

    Image Alt Text: Descriptive for accessibility and SEO
